2. 산업 동향: 시장 규모, CAGR, 트렌드

ws_assum.title = 'Assumptions'

# Sheet 2-4: IS, BS, CF (3-Statement 연결)
ws_is = wb.create_sheet('IS')
ws_bs = wb.create_sheet('BS')
ws_cf = wb.create_sheet('CF')

# Sheet 5: Valuation (PER/PBR/EV_EBITDA/DCF)
ws_val = wb.create_sheet('Valuation')

# Sheet 6: Scenario (Bear/Base/Bull)
ws_scen = wb.create_sheet('Scenario')

# Sheet 7: DCF Sensitivity
ws_dcf = wb.create_sheet('DCF_Sensitivity')

# Sheet 8: Peer Comparison
ws_peer = wb.create_sheet('Peer')

# Sheet 9: Summary
ws_sum = wb.create_sheet('Summary')

# ... 각 시트에 데이터 + 서식 적용
